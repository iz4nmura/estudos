#include <stdio.h>

struct Jogador {
    int numero;
    float peso;
    float altura;
    char inicial;
};

int main() {
    struct Jogador jogadores[3];

    
    for (int i = 0; i < 3; i++) {
        printf("Jogador %d:\n", i + 1);

        printf("Numero da camisa: ");
        scanf("%d", &jogadores[i].numero);

        printf("Peso: ");
        scanf("%f", &jogadores[i].peso);

        printf("Altura: ");
        scanf("%f", &jogadores[i].altura);

        printf("Inicial do nome: ");
        scanf(" %c", &jogadores[i].inicial);

        printf("\n");
    }

    
    int menorAlturaIndex = 0;
    for (int i = 1; i < 3; i++) {
        if (jogadores[i].altura < jogadores[menorAlturaIndex].altura) {
            menorAlturaIndex = i;
        }
    }

    
    int maiorPesoIndex = 0;
    for (int i = 1; i < 3; i++) {
        if (jogadores[i].peso > jogadores[maiorPesoIndex].peso) {
            maiorPesoIndex = i;
        }
    }

    
    printf("Inicial do jogador mais baixo: %c\n", jogadores[menorAlturaIndex].inicial);
    printf("Numero do jogador mais pesado: %d\n", jogadores[maiorPesoIndex].numero);

    return 0;
}
